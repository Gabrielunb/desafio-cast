CREATE SCHEMA IF NOT EXISTS DesafioCast;
USE DesafioCast;

CREATE TABLE IF NOT EXISTS Categorias(
	codigo_categoria int primary key auto_increment,
	descricao varchar(50) not null
);

CREATE TABLE IF NOT EXISTS Curso(
	codigo_curso int primary key auto_increment,
	data_inicio datetime not null,
	data_termino datetime not null,
	qtd_turma int,
	categoria_codigo int,
	FOREIGN KEY (categoria_codigo) REFERENCES Categorias(codigo_categoria)
);
